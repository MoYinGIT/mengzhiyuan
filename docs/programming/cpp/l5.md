## 📚 模块一：数论基础

### 1.1 最大公约数（GCD）

**定义**：能同时整除两个数的最大正整数

**欧几里得算法**：
```cpp
// 递归版
int gcd(int a, int b) {
    if (b == 0) return a;
    return gcd(b, a % b);
}

// 迭代版
int gcd(int a, int b) {
    while (b != 0) {
        int temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}
```

**C++11 自己实现**（推荐，竞赛常用）：
```cpp
int gcd(int a, int b) {
    while (b != 0) {
        int temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

// 使用
int result = gcd(48, 18);  // 6
```

**C++17 标准库**（如果编译器支持）：
```cpp
#include <numeric>
int result = std::gcd(48, 18);  // 6
```

### 1.2 最小公倍数（LCM）

**公式**：`lcm(a, b) = a * b / gcd(a, b)`

```cpp
int lcm(int a, int b) {
    return a / gcd(a, b) * b;  // 先除后乘，防止溢出
}
```

### 1.3 素数判断

**试除法**（O(√n)）：
```cpp
bool is_prime(int n) {
    if (n <= 1) return false;
    if (n <= 3) return true;
    if (n % 2 == 0 || n % 3 == 0) return false;
    
    for (int i = 5; i * i <= n; i += 6) {
        if (n % i == 0 || n % (i + 2) == 0) {
            return false;
        }
    }
    return true;
}
```

### 1.4 素数筛

**埃氏筛**（O(n log log n)）：
```cpp
const int MAX = 1000000;
bool is_prime[MAX + 1];

void sieve(int n) {
    // 初始化
    for (int i = 0; i <= n; i++) {
        is_prime[i] = true;
    }
    is_prime[0] = is_prime[1] = false;
    
    // 筛
    for (int i = 2; i * i <= n; i++) {
        if (is_prime[i]) {
            for (int j = i * i; j <= n; j += i) {
                is_prime[j] = false;
            }
        }
    }
}

// 使用
sieve(100);
if (is_prime[97]) cout << "97是素数";
```

### 1.5 分解质因数

```cpp
void prime_factorization(int n) {
    cout << n << " = ";
    bool first = true;
    
    for (int i = 2; i * i <= n; i++) {
        while (n % i == 0) {
            if (!first) cout << " × ";
            cout << i;
            first = false;
            n /= i;
        }
    }
    
    if (n > 1) {
        if (!first) cout << " × ";
        cout << n;
    }
    cout << endl;
}

// 60 = 2 × 2 × 3 × 5
```

## 📚 模块二：链表

### 2.1 什么是链表？

**链表**：通过指针连接的节点序列

**对比**：
| 特性 | 数组 | 链表 |
|:---|:---|:---|
| 存储 | 连续 | 分散 |
| 插入删除 | 慢（需要移动） | 快（改指针） |
| 随机访问 | O(1) | O(n) |
| 大小 | 固定 | 动态 |

### 2.2 单链表实现

```cpp
#include <iostream>
using namespace std;

// 节点结构
struct Node {
    int data;
    Node* next;
};

// 创建节点
Node* create_node(int val) {
    Node* new_node = new Node();
    new_node->data = val;
    new_node->next = nullptr;
    return new_node;
}

// 头插法
void insert_head(Node* &head, int val) {
    Node* new_node = create_node(val);
    new_node->next = head;
    head = new_node;
}

// 尾插法
void insert_tail(Node* &head, int val) {
    Node* new_node = create_node(val);
    if (head == nullptr) {
        head = new_node;
        return;
    }
    Node* temp = head;
    while (temp->next != nullptr) {
        temp = temp->next;
    }
    temp->next = new_node;
}

// 删除节点
void delete_node(Node* &head, int val) {
    if (head == nullptr) return;
    
    if (head->data == val) {
        Node* temp = head;
        head = head->next;
        delete temp;
        return;
    }
    
    Node* current = head;
    while (current->next != nullptr && current->next->data != val) {
        current = current->next;
    }
    
    if (current->next != nullptr) {
        Node* temp = current->next;
        current->next = current->next->next;
        delete temp;
    }
}

// 打印链表
void print_list(Node* head) {
    Node* temp = head;
    while (temp != nullptr) {
        cout << temp->data;
        if (temp->next != nullptr) cout << " -> ";
        temp = temp->next;
    }
    cout << " -> NULL" << endl;
}

// 释放内存
void free_list(Node* head) {
    while (head != nullptr) {
        Node* temp = head;
        head = head->next;
        delete temp;
    }
}

int main() {
    Node* head = nullptr;
    
    insert_tail(head, 1);
    insert_tail(head, 2);
    insert_tail(head, 3);
    insert_head(head, 0);
    
    print_list(head);  // 0 -> 1 -> 2 -> 3 -> NULL
    
    delete_node(head, 2);
    print_list(head);  // 0 -> 1 -> 3 -> NULL
    
    free_list(head);
    return 0;
}
```

### 2.3 链表反转

```cpp
Node* reverse(Node* head) {
    Node* prev = nullptr;
    Node* current = head;
    Node* next = nullptr;
    
    while (current != nullptr) {
        next = current->next;      // 保存下一个
        current->next = prev;       // 反转指针
        prev = current;             // 前移prev
        current = next;             // 前移current
    }
    
    return prev;  // 新的头节点
}
```

## 📚 模块三：分治算法

### 3.1 分治思想

**分治 = 分而治之**：
1. **分**：将问题分解为子问题
2. **治**：解决子问题
3. **合**：合并子问题的解

### 3.2 归并排序（分治典型）

```cpp
void merge(int arr[], int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;
    
    int L[n1], R[n2];
    for (int i = 0; i < n1; i++) L[i] = arr[left + i];
    for (int i = 0; i < n2; i++) R[i] = arr[mid + 1 + i];
    
    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) arr[k++] = L[i++];
        else arr[k++] = R[j++];
    }
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

void merge_sort(int arr[], int left, int right) {
    if (left >= right) return;
    
    int mid = left + (right - left) / 2;
    
    merge_sort(arr, left, mid);       // 分：排序左半
    merge_sort(arr, mid + 1, right);  // 分：排序右半
    merge(arr, left, mid, right);     // 合：合并
}
```

### 3.3 快速幂

**计算 a^n**：

```cpp
// 朴素方法：O(n)
int power_naive(int a, int n) {
    int result = 1;
    for (int i = 0; i < n; i++) {
        result *= a;
    }
    return result;
}

// 快速幂：O(log n)
long long power(long long a, long long n, long long mod) {
    long long result = 1;
    a = a % mod;
    while (n > 0) {
        if (n & 1)  // 如果n是奇数
            result = result * a % mod;
        a = a * a % mod;
        n >>= 1;  // n /= 2
    }
    return result;
}

// 计算 2^100 mod 1000000007
// power(2, 100, 1000000007)
```

## 🎮 综合项目：学生管理系统 V3

结合结构体、文件操作、链表的综合项目：

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

struct Student {
    int id;
    string name;
    int age;
    double score;
    Student* next;
};

class StudentManager {
    Student* head;
    int count;
    
public:
    StudentManager() : head(nullptr), count(0) {}
    
    // 添加学生
    void add(int id, string name, int age, double score) {
        Student* s = new Student{id, name, age, score, head};
        head = s;
        count++;
    }
    
    // 查找
    Student* find(int id) {
        Student* p = head;
        while (p != nullptr) {
            if (p->id == id) return p;
            p = p->next;
        }
        return nullptr;
    }
    
    // 删除
    void remove(int id) {
        if (head == nullptr) return;
        
        if (head->id == id) {
            Student* temp = head;
            head = head->next;
            delete temp;
            count--;
            return;
        }
        
        Student* p = head;
        while (p->next != nullptr && p->next->id != id) {
            p = p->next;
        }
        
        if (p->next != nullptr) {
            Student* temp = p->next;
            p->next = p->next->next;
            delete temp;
            count--;
        }
    }
    
    // 显示所有
    void display() {
        Student* p = head;
        cout << "ID\t姓名\t年龄\t成绩\n";
        while (p != nullptr) {
            cout << p->id << "\t"
                 << p->name << "\t"
                 << p->age << "\t"
                 << p->score << endl;
            p = p->next;
        }
    }
    
    // 保存到文件
    void save(string filename) {
        ofstream file(filename);
        Student* p = head;
        while (p != nullptr) {
            file << p->id << " "
                 << p->name << " "
                 << p->age << " "
                 << p->score << endl;
            p = p->next;
        }
        file.close();
    }
    
    // 析构函数（释放内存）
    ~StudentManager() {
        while (head != nullptr) {
            Student* temp = head;
            head = head->next;
            delete temp;
        }
    }
};

int main() {
    StudentManager sm;
    
    sm.add(1, "张三", 12, 89.5);
    sm.add(2, "李四", 11, 92.0);
    sm.add(3, "王五", 13, 85.5);
    
    cout << "所有学生：\n";
    sm.display();
    
    sm.save("students.txt");
    cout << "\n已保存到 students.txt\n";
    
    return 0;
}
```

## 🏛️ 三经典智慧

### 《易经》· 生生不息

**链表就是生生不息**：
- 每个节点指向下一个
- 无穷无尽，直到 NULL
- 插入删除，生生不息

### 《道德经》· 有之以为利，无之以为用

**链表的有无之道**：
- 有：节点的数据
- 无：next 指针的"空"
- 无之以为用：NULL 表示结束

### 《论语》· 学而不思则罔

**学习链表**：
- 画图理解指针变化
- 动手写代码，不要只看
- 调试跟踪每个节点的变化
