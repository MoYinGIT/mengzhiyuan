```cpp
#include <iostream>

    for (int i = 1; i <= 10; i++) {
        sum += i;
    }
    cout << "1到10的和=" << sum << endl;
```

```cpp
#include <iostream>
```

```cpp
int main() {
    // 错误1：死循环（条件永远为真）
    // for (int i = 0; i >= 0; i++) { ... }  // i永远>=0，死循环
    
    // 错误2：数组越界
    int arr[5] = {1, 2, 3, 4, 5};
    // for (int i = 0; i <= 5; i++) { ... }  // i=5时越界
    
    // 错误3：忘记更新循环变量
    // for (int i = 0; i < 5; ) { ... }  // 如果循环体内不更新i，死循环
    
    // 正确写法
    cout << "正确写法：";
    for (int i = 0; i < 5; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
    
    // 错误4：off-by-one错误
    int sum = 0;
    for (int i = 1; i <= 10; i++) {  // 正确的：1到10
        sum += i;
    }
    cout << "1到10的和=" << sum << endl;
    
    return 0;
}
```

## 【大题10】for循环的综合应用

**考察点**：for循环的综合应用和优化技巧

```cpp
#include <iostream>
using namespace std;

int main() {
    // 应用1：打印金字塔
    int rows = 5;
    for (int i = 1; i <= rows; i++) {
        // 打印空格
        for (int j = 1; j <= rows - i; j++) {
            cout << " ";
        }
        // 打印星号
        for (int k = 1; k <= 2 * i - 1; k++) {
            cout << "*";
        }
        cout << endl;
    }
    
    // 应用2：判断素数
    int num = 17;
    bool isPrime = true;
    for (int i = 2; i * i <= num; i++) {
        if (num % i == 0) {
            isPrime = false;
            break;
        }
    }
    if (isPrime) cout << num << "是素数" << endl;
    else cout << num << "不是素数" << endl;
    
    // 应用3：优化循环（只遍历到平方根）
    int count = 0;
    for (int i = 1; i * i <= 100; i++) {
        count++;
    }
    cout << "1到10的平方数个数：" << count << endl;
    
    return 0;
}
```

## 核心要点总结

### for循环使用检查清单

- [ ] 基本格式：`for (初始化; 条件; 更新) { 循环体; }`
- [ ] 执行顺序：初始化 → 条件 → 循环体 → 更新 → 条件...
- [ ] 三个部分都可以省略，但分号不能省略
- [ ] 累加初始值设为0，累乘初始值设为1
- [ ] 遍历数组：i从0开始，i < n结束
- [ ] 嵌套循环：内层循环体执行m×n次
- [ ] break跳出整个循环，continue跳过本次循环
- [ ] 避免死循环：确保条件最终会变为假
- [ ] 避免数组越界：i < n而不是i <= n

### 常见错误对比

| 错误 | 问题 | 正确做法 |
|:---|:---|:---|
| `i <= n`遍历数组 | 数组越界 | 用`i < n` |
| `i >= 0; i++` | 死循环 | 确保条件能变为假 |
| 忘记更新循环变量 | 死循环 | 在更新部分或循环体内更新 |
| 累乘初始化为0 | 结果永远为0 | 初始化为1 |
