## 📝 代码实现

### C++版本

```cpp
/**
 * A1258: 导弹拦截
 * 题目要求：
 * 1. 一套系统最多能拦截多少导弹（最长不上升子序列）
 * 2. 要拦截所有导弹最少需要多少套系统（Dilworth定理：最少链划分=最长反链长度）
 * 
 * 问题分析：之前TLE是因为O(n^2)算法，需要优化到O(n log n)
 * 解法：
 * 1. 第一问：最长不上升子序列 - 用二分优化
 * 2. 第二问：最长上升子序列 - 用二分优化（Dilworth定理）
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    vector&lt;int&gt; a;
    int x;
    // 读取直到EOF
    while (cin &gt;&gt; x) {
        a.push_back(x);
    }
    
    int n = a.size();
    if (n == 0) {
        cout &lt;&lt; 0 &lt;&lt; endl &lt;&lt; 0 &lt;&lt; endl;
        return 0;
    }
    
    // 第一问：最长不上升子序列（用二分优化 O(n log n)）
    // d[i]: 长度为i的不上升子序列的最大结尾元素
    vector&lt;int&gt; d1;
    for (int i = 0; i &lt; n; i++) {
        // 找第一个小于a[i]的位置（不上升：允许等于，所以找&lt;）
        auto it = upper_bound(d1.begin(), d1.end(), a[i], greater&lt;int&gt;());
        if (it == d1.end()) {
            d1.push_back(a[i]);
        } else {
            *it = a[i];
        }
    }
    
    // 第二问：最少系统数 = 最长上升子序列长度（Dilworth定理）
    // 用二分优化求LIS
    vector&lt;int&gt; d2;
    for (int i = 0; i &lt; n; i++) {
        auto it = lower_bound(d2.begin(), d2.end(), a[i]);
        if (it == d2.end()) {
            d2.push_back(a[i]);
        } else {
            *it = a[i];
        }
    }
    
    cout &lt;&lt; d1.size() &lt;&lt; endl;
    cout &lt;&lt; d2.size() &lt;&lt; endl;
    
    return 0;
}

```
