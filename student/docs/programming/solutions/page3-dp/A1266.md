```cpp
/**
 * A1266: 完全背包问题
 * 每个物品可以选无限次
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin &gt;&gt; m &gt;&gt; n;
    
    vector&lt;int&gt; w(n), v(n);
    for (int i = 0; i &lt; n; i++) {
        cin &gt;&gt; w[i] &gt;&gt; v[i];
    }
    
    // dp[j]: 容量为j时的最大价值
    vector&lt;int&gt; dp(m + 1, 0);
    
    for (int i = 0; i &lt; n; i++) {
        for (int j = w[i]; j &lt;= m; j++) {  // 正序遍历
            dp[j] = max(dp[j], dp[j - w[i]] + v[i]);
        }
    }
    
    cout &lt;&lt; dp[m] &lt;&lt; endl;
    
    return 0;
}

```
