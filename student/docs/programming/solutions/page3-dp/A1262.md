## 📝 代码实现

### C++版本

```cpp
/**
 * A1262: 过河卒
 * 棋盘DP，考虑马控制点
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

typedef long long ll;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m, mx, my;
    cin &gt;&gt; n &gt;&gt; m &gt;&gt; mx &gt;&gt; my;
    
    // 马控制点
    bool blocked[25][25] = {false};
    int dx[] = {-2, -2, -1, -1, 1, 1, 2, 2, 0};
    int dy[] = {-1, 1, -2, 2, -2, 2, -1, 1, 0};
    
    for (int i = 0; i &lt; 9; i++) {
        int nx = mx + dx[i];
        int ny = my + dy[i];
        if (nx &gt;= 0 && nx &lt;= n && ny &gt;= 0 && ny &lt;= m) {
            blocked[nx][ny] = true;
        }
    }
    
    // dp[i][j]: 从(0,0)到(i,j)的路径数
    vector&lt;vector&lt;ll&gt;&gt; dp(n + 1, vector&lt;ll&gt;(m + 1, 0));
    dp[0][0] = blocked[0][0] ? 0 : 1;
    
    for (int i = 0; i &lt;= n; i++) {
        for (int j = 0; j &lt;= m; j++) {
            if (blocked[i][j] || (i == 0 && j == 0)) continue;
            if (i &gt; 0) dp[i][j] += dp[i-1][j];
            if (j &gt; 0) dp[i][j] += dp[i][j-1];
        }
    }
    
    cout &lt;&lt; dp[n][m] &lt;&lt; endl;
    
    return 0;
}

```
