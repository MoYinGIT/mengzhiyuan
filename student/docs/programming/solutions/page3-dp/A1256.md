## 📝 代码实现

### C++版本

```cpp
/**
 * A1256: 数字金字塔
 * DP基础题，从顶向下或从下向上递推
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin &gt;&gt; n;
    
    vector&lt;vector&lt;int&gt;&gt; a(n, vector&lt;int&gt;(n));
    for (int i = 0; i &lt; n; i++) {
        for (int j = 0; j &lt;= i; j++) {
            cin &gt;&gt; a[i][j];
        }
    }
    
    // dp[i][j]: 从底向上，到(i,j)的最大路径和
    vector&lt;vector&lt;int&gt;&gt; dp = a;
    
    // 从下向上递推
    for (int i = n - 2; i &gt;= 0; i--) {
        for (int j = 0; j &lt;= i; j++) {
            dp[i][j] += max(dp[i+1][j], dp[i+1][j+1]);
        }
    }
    
    cout &lt;&gt; dp[0][0] &lt;&lt; endl;
    
    return 0;
}

```
