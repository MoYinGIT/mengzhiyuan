/**
 * A1275: 金明的预算方案
 * 分组背包问题（主件和附件）
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin &gt;&gt; n &gt;&gt; m;
    
    vector&lt;int&gt; v(m), p(m), q(m);  // 价格、重要度、主件编号
    for (int i = 0; i &lt; m; i++) {
        cin &gt;&gt; v[i] &gt;&gt; p[i] &gt;&gt; q[i];
    }
    
    // dp[j]: 预算为j时的最大价值
    vector&lt;int&gt; dp(n + 1, 0);
    
    // 先处理主件
    for (int i = 0; i &lt; m; i++) {
        if (q[i] == 0) {  // 主件
            for (int j = n; j &gt;= v[i]; j--) {
                // 只选主件
                dp[j] = max(dp[j], dp[j - v[i]] + v[i] * p[i]);
                
                // 找附件并尝试组合
                for (int k = 0; k &lt; m; k++) {
                    if (q[k] == i + 1) {  // k是i的附件
                        if (j &gt;= v[i] + v[k]) {
                            dp[j] = max(dp[j], dp[j - v[i] - v[k]] + v[i] * p[i] + v[k] * p[k]);
                        }
                    }
                }
            }
        }
    }
    
    cout &lt;&lt; dp[n] &lt;&lt; endl;
    
    return 0;
}
