## 📝 代码实现

### C++版本

```cpp
/**
 * A1271: 潜水员
 * 二维费用背包
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

const int INF = 0x3f3f3f3f;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int m, n, k;
    cin &gt;&gt; m &gt;&gt; n &gt;&gt; k;
    
    // dp[i][j]: 氧气至少为i，氮气至少为j时的最小重量
    // 注意：可能需要更多空间
    vector&lt;vector&lt;int&gt;&gt; dp(m + 100, vector&lt;int&gt;(n + 100, INF));
    dp[0][0] = 0;
    
    for (int t = 0; t &lt; k; t++) {
        int o, ni, w;
        cin &gt;&gt; o &gt;&gt; ni &gt;&gt; w;
        
        for (int i = m; i &gt;= 0; i--) {
            for (int j = n; j &gt;= 0; j--) {
                int ni2 = min(m, i + o);
                int nj2 = min(n, j + ni);
                dp[ni2][nj2] = min(dp[ni2][nj2], dp[i][j] + w);
            }
        }
    }
    
    cout &lt;&lt; dp[m][n] &lt;&lt; endl;
    
    return 0;
}

```
