## 📝 代码实现

### C++版本

```cpp
/**
 * A1277: 没有上司的舞会
 * 树形DP经典题
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

vector&lt;vector&lt;int&gt;&gt; children;
vector&lt;int&gt; r;
vector&lt;vector&lt;long long&gt;&gt; dp;  // dp[u][0/1]: u不参加/参加的最大快乐值

void solve(int u) {
    dp[u][0] = 0;
    dp[u][1] = r[u];
    
    for (int v : children[u]) {
        solve(v);
        dp[u][0] += max(dp[v][0], dp[v][1]);
        dp[u][1] += dp[v][0];
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin &gt;&gt; n;
    
    r.resize(n);
    for (int i = 0; i &lt; n; i++) {
        cin &gt;&gt; r[i];
    }
    
    children.resize(n);
    vector&lt;bool&gt; hasBoss(n, false);
    
    while (true) {
        int l, k;
        cin &gt;&gt; l &gt;&gt; k;
        if (l == 0 && k == 0) break;
        l--; k--;
        children[k].push_back(l);
        hasBoss[l] = true;
    }
    
    // 找根
    int root = 0;
    while (root &lt; n && hasBoss[root]) root++;
    
    dp.resize(n, vector&lt;long long&gt;(2));
    solve(root);
    
    cout &lt;&lt; max(dp[root][0], dp[root][1]) &lt;&lt; endl;
    
    return 0;
}

```
