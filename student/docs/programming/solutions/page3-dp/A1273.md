#include &lt;bits/stdc++.h&gt;
using namespace std;

typedef long long ll;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin &gt;&gt; n;
    
    // 根据题目具体要求实现
    // 可能是从顶部放下小球，经过钉子的路径计数
    
    vector&lt;vector&lt;ll&gt;&gt; dp(n + 2, vector&lt;ll&gt;(n + 2, 0));
    dp[1][1] = 1;
    
    for (int i = 1; i &lt;= n; i++) {
        for (int j = 1; j &lt;= i; j++) {
            if (dp[i][j] &gt; 0) {
                // 钉子会使小球左右分开
                dp[i+1][j] += dp[i][j];
                dp[i+1][j+1] += dp[i][j];
            }
        }
    }
    
    // 输出底部某个位置的路径数
    // 具体输出格式根据题目要求
    
    return 0;
}
