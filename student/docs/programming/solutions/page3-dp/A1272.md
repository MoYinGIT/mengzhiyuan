/**
 * A1272: 编辑距离
 * 经典DP，求两个字符串的编辑距离
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    string s1, s2;
    cin &gt;&gt; s1 &gt;&gt; s2;
    
    int n = s1.length();
    int m = s2.length();
    
    // dp[i][j]: s1[0..i-1]变成s2[0..j-1]的最小操作次数
    vector&lt;vector&lt;int&gt;&gt; dp(n + 1, vector&lt;int&gt;(m + 1));
    
    // 初始化
    for (int i = 0; i &lt;= n; i++) dp[i][0] = i;  // 删除i个字符
    for (int j = 0; j &lt;= m; j++) dp[0][j] = j;  // 插入j个字符
    
    for (int i = 1; i &lt;= n; i++) {
        for (int j = 1; j &lt;= m; j++) {
            if (s1[i-1] == s2[j-1]) {
                dp[i][j] = dp[i-1][j-1];
            } else {
                dp[i][j] = min({dp[i-1][j], dp[i][j-1], dp[i-1][j-1]}) + 1;
                // 删除      插入       替换
            }
        }
    }
    
    cout &lt;&lt; dp[n][m] &lt;&lt; endl;
    
    return 0;
}
