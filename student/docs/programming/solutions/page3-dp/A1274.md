```cpp
/**
 * A1274: 能量项链
 * 区间DP，类似矩阵链乘
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin &gt;&gt; n;
    
    vector&lt;int&gt; a(n);
    for (int i = 0; i &lt; n; i++) {
        cin &gt;&gt; a[i];
    }
    
    // 复制数组形成环
    vector&lt;int&gt; b(2 * n);
    for (int i = 0; i &lt; 2 * n; i++) {
        b[i] = a[i % n];
    }
    
    // dp[i][j]: 合并区间[i,j]的最大能量
    vector&lt;vector&lt;long long&gt;&gt; dp(2 * n, vector&lt;long long&gt;(2 * n, 0));
    
    for (int len = 2; len &lt;= n; len++) {
        for (int i = 0; i + len &lt; 2 * n; i++) {
            int j = i + len;
            for (int k = i + 1; k &lt; j; k++) {
                dp[i][j] = max(dp[i][j], dp[i][k] + dp[k][j] + 1LL * b[i] * b[k] * b[j]);
            }
        }
    }
    
    long long ans = 0;
    for (int i = 0; i &lt; n; i++) {
        ans = max(ans, dp[i][i + n]);
    }
    
    cout &lt;&lt; ans &lt;&lt; endl;
    
    return 0;
}

```
