## 📝 代码实现

### C++版本

```cpp
/**
 * A1257: 最长不下降序列
 * 题目要求：输出最长不下降子序列的长度和具体序列
 * 
 * 问题分析：之前WA是因为只输出了长度，没有输出具体序列
 * 解法：DP求LIS，同时记录路径
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin &gt;&gt; n;
    
    vector&lt;int&gt; a(n);
    for (int i = 0; i &lt; n; i++) {
        cin &gt;&gt; a[i];
    }
    
    // dp[i]: 以a[i]结尾的最长不下降子序列长度
    vector&lt;int&gt; dp(n, 1);
    // pre[i]: 记录前驱节点，用于还原路径
    vector&lt;int&gt; pre(n, -1);
    
    int maxLen = 1, maxIdx = 0;
    
    for (int i = 1; i &lt; n; i++) {
        for (int j = 0; j &lt; i; j++) {
            if (a[j] &lt;= a[i] && dp[j] + 1 &gt; dp[i]) {
                dp[i] = dp[j] + 1;
                pre[i] = j;
            }
        }
        if (dp[i] &gt; maxLen) {
            maxLen = dp[i];
            maxIdx = i;
        }
    }
    
    // 还原路径
    vector&lt;int&gt; path;
    int cur = maxIdx;
    while (cur != -1) {
        path.push_back(a[cur]);
        cur = pre[cur];
    }
    
    // 输出结果：先输出长度，然后输出序列（从前往后）
    cout &lt;&lt; maxLen &lt;&lt; endl;
    for (int i = path.size() - 1; i &gt;= 0; i--) {
        if (i &lt; path.size() - 1) cout &lt;&lt; " ";
        cout &lt;&lt; path[i];
    }
    cout &lt;&lt; endl;
    
    return 0;
}

```
