```cpp
/**
 * A1270: 装箱问题
 * 01背包，最小剩余空间 = 容量 - 最大装载
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int v, n;
    cin &gt;&gt; v &gt;&gt; n;
    
    vector&lt;int&gt; w(n);
    for (int i = 0; i &lt; n; i++) {
        cin &gt;&gt; w[i];
    }
    
    // dp[j]: 容量为j时能否装满
    vector&lt;bool&gt; dp(v + 1, false);
    dp[0] = true;
    
    for (int i = 0; i &lt; n; i++) {
        for (int j = v; j &gt;= w[i]; j--) {
            if (dp[j - w[i]]) dp[j] = true;
        }
    }
    
    // 找最大能装满的容量
    int maxLoad = 0;
    for (int j = v; j &gt;= 0; j--) {
        if (dp[j]) {
            maxLoad = j;
            break;
        }
    }
    
    cout &lt;&lt; v - maxLoad &lt;&lt; endl;
    
    return 0;
}

```
