## 📝 代码实现

### C++版本

```cpp
/**
 * A1276: 树形DP入门
 * 树上最大独立集
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

vector&lt;vector&lt;int&gt;&gt; tree;
vector&lt;int&gt; happy;
vector&lt;vector&lt;long long&gt;&gt; dp;  // dp[u][0/1]: u不选/选的最大快乐值

void dfs(int u, int parent) {
    dp[u][0] = 0;
    dp[u][1] = happy[u];
    
    for (int v : tree[u]) {
        if (v == parent) continue;
        dfs(v, u);
        
        // u不选，子节点可选可不选
        dp[u][0] += max(dp[v][0], dp[v][1]);
        // u选，子节点不能选
        dp[u][1] += dp[v][0];
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin &gt;&gt; n;
    
    happy.resize(n);
    for (int i = 0; i &lt; n; i++) {
        cin &gt;&gt; happy[i];
    }
    
    tree.resize(n);
    vector&lt;bool&gt; isChild(n, false);
    
    for (int i = 0; i &lt; n - 1; i++) {
        int l, k;
        cin &gt;&gt; l &gt;&gt; k;
        l--; k--;  // 转为0-indexed
        tree[k].push_back(l);
        isChild[l] = true;
    }
    
    // 找根节点
    int root = 0;
    while (root &lt; n && isChild[root]) root++;
    
    dp.resize(n, vector&lt;long long&gt;(2, 0));
    dfs(root, -1);
    
    cout &lt;&lt; max(dp[root][0], dp[root][1]) &lt;&lt; endl;
    
    return 0;
}

```
