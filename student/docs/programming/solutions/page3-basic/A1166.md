## 📝 代码实现

### C语言版本

```c
/*
 * 题目24：A1166 求f(x,n)
 * 
 * f(x,n) = 
 *   sqrt(x)                 (n = 1)
 *   sqrt(n + f(x, n-1))     (n &gt; 1)
 * 
 * 嵌套平方根公式
 */

#include &lt;stdio.h&gt;
#include &lt;math.h&gt;

double f(double x, int n) {
    if (n == 1) {
        return sqrt(x);
    }
    return sqrt(n + f(x, n - 1));
}

int main() {
    double x;
    int n;
    scanf("%lf %d", &x, &n);
    printf("%.2f\n", f(x, n));
    return 0;
}

```
