## 📝 代码实现

### C++版本

```cpp
/*
 * 题目8：A1147 最高分数的学生姓名
 * 
 * 输入n个学生的姓名和分数，找出分数最高的学生
 */

#include &lt;iostream&gt;
#include &lt;string&gt;
using namespace std;

int main() {
    int n;
    cin &gt;&gt; n;
    
    string name[100];
    int score[100];
    
    for (int i = 0; i &lt; n; i++) {
        cin &gt;&gt; name[i] &gt;&gt; score[i];
    }
    
    int max_idx = 0;
    for (int i = 1; i &lt; n; i++) {
        if (score[i] &gt; score[max_idx]) {
            max_idx = i;
        }
    }
    
    cout &lt;&lt; name[max_idx] &lt;&lt; endl;
    
    return 0;
}

```
