## 📝 代码实现

### C语言版本

```c
/*
 * 题目21：A1163 阿克曼(Ackmann)函数
 * 
 * Ack(m,n) = 
 *   n + 1                    (m = 0)
 *   Ack(m-1, 1)              (m &gt; 0, n = 0)
 *   Ack(m-1, Ack(m, n-1))    (m &gt; 0, n &gt; 0)
 */

#include &lt;stdio.h&gt;

int ack(int m, int n) {
    if (m == 0) {
        return n + 1;
    } else if (n == 0) {
        return ack(m - 1, 1);
    } else {
        return ack(m - 1, ack(m, n - 1));
    }
}

int main() {
    int m, n;
    scanf("%d %d", &m, &n);
    printf("%d\n", ack(m, n));
    return 0;
}

```
