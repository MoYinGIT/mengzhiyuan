## 📝 代码实现

### C语言版本

```c
/*
 * 题目5：A1127 图像旋转
 * 
 * 将图像顺时针旋转90度
 * 
 * 算法：新图像的第i行第j列 = 原图像的第(n-1-j)行第i列
 * 或者理解为：顺时针90度旋转 = 先转置再左右翻转
 */

#include &lt;stdio.h&gt;

int main() {
    int n, m;
    int a[100][100];
    
    scanf("%d %d", &n, &m);
    
    for (int i = 0; i &lt; n; i++) {
        for (int j = 0; j &lt; m; j++) {
            scanf("%d", &a[i][j]);
        }
    }
    
    // 顺时针旋转90度：新矩阵是m×n
    // 新矩阵[i][j] = 原矩阵[n-1-j][i]
    for (int i = 0; i &lt; m; i++) {
        for (int j = 0; j &lt; n; j++) {
            if (j &gt; 0) printf(" ");
            printf("%d", a[n - 1 - j][i]);
        }
        printf("\n");
    }
    
    return 0;
}

```
