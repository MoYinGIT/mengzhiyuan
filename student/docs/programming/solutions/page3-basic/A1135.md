/*
 * 题目7：A1135 配对碱基链
 * 
 * DNA碱基配对规则：A-T, T-A, C-G, G-C
 */

#include &lt;iostream&gt;
#include &lt;string&gt;
using namespace std;

int main() {
    string s;
    cin &gt;&gt; s;
    
    for (char c : s) {
        switch (c) {
            case 'A': cout &lt;&lt; 'T'; break;
            case 'T': cout &lt;&lt; 'A'; break;
            case 'C': cout &lt;&lt; 'G'; break;
            case 'G': cout &lt;&lt; 'C'; break;
        }
    }
    cout &lt;&lt; endl;
    
    return 0;
}
