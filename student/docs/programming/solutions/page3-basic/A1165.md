## 📝 代码实现

### C++版本

```cpp
/*
 * 题目23：A1165 Hermite多项式
 */

#include &lt;iostream&gt;
#include &lt;cmath&gt;
using namespace std;

double hermite(int n, double x) {
    if (n == 0) return 1;
    if (n == 1) return 2 * x;
    return 2 * x * hermite(n - 1, x) - 2 * (n - 1) * hermite(n - 2, x);
}

int main() {
    int n;
    double x;
    cin &gt;&gt; n &gt;&gt; x;
    printf("%.2f\n", hermite(n, x));
    return 0;
}

```
