## 📝 代码实现

### C++版本

/*
 * 题目10：A1152 最大数max(x,y,z)
 * 
 * 自定义函数求三个数的最大值
 */

#include &lt;iostream&gt;
using namespace std;

int max(int x, int y, int z) {
    int m = x;
    if (y &gt; m) m = y;
    if (z &gt; m) m = z;
    return m;
}

int main() {
    int x, y, z;
    cin &gt;&gt; x &gt;&gt; y &gt;&gt; z;
    cout &lt;&lt; max(x, y, z) &lt;&lt; endl;
    return 0;
}
