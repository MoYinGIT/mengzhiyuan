## 📝 代码实现

### C语言版本

```c
/*
 * 题目15：A1157 判决素数个数
 * 
 * 统计区间[m,n]内有多少个素数
 */

#include &lt;stdio.h&gt;

int is_prime(int n) {
    if (n &lt; 2) return 0;
    if (n == 2) return 1;
    if (n % 2 == 0) return 0;
    for (int i = 3; i * i &lt;= n; i += 2) {
        if (n % i == 0) return 0;
    }
    return 1;
}

int main() {
    int m, n;
    scanf("%d %d", &m, &n);
    
    int count = 0;
    for (int i = m; i &lt;= n; i++) {
        if (is_prime(i)) {
            count++;
        }
    }
    
    printf("%d\n", count);
    
    return 0;
}

```
