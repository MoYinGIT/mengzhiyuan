## 📝 代码实现

### C++版本

```cpp
/*
 * 题目11：A1153 绝对素数
 * 
 * 绝对素数：一个素数，其逆序也是素数（且不等于原数）
 */

#include &lt;iostream&gt;
using namespace std;

int is_prime(int n) {
    if (n &lt; 2) return 0;
    if (n == 2) return 1;
    if (n % 2 == 0) return 0;
    for (int i = 3; i * i &lt;= n; i += 2) {
        if (n % i == 0) return 0;
    }
    return 1;
}

int reverse(int n) {
    int rev = 0;
    while (n &gt; 0) {
        rev = rev * 10 + n % 10;
        n /= 10;
    }
    return rev;
}

int main() {
    int n;
    cin &gt;&gt; n;
    
    int rev = reverse(n);
    
    if (is_prime(n) && is_prime(rev) && n != rev) {
        cout &lt;&lt; "Yes" &lt;&lt; endl;
    } else {
        cout &lt;&lt; "No" &lt;&lt; endl;
    }
    
    return 0;
}

```
