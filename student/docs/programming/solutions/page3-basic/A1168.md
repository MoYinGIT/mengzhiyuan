## 📝 代码实现

### C++版本

```cpp
/*
 * 题目26：A1168 大整数加法
 */

#include &lt;iostream&gt;
#include &lt;string&gt;
using namespace std;

int main() {
    string a, b;
    int res[300];
    
    cin &gt;&gt; a &gt;&gt; b;
    
    int len_a = a.length();
    int len_b = b.length();
    
    int i = len_a - 1, j = len_b - 1;
    int carry = 0;
    int k = 0;
    
    while (i &gt;= 0 || j &gt;= 0 || carry) {
        int sum = carry;
        if (i &gt;= 0) sum += a[i--] - '0';
        if (j &gt;= 0) sum += b[j--] - '0';
        res[k++] = sum % 10;
        carry = sum / 10;
    }
    
    for (int idx = k - 1; idx &gt;= 0; idx--) {
        cout &lt;&lt; res[idx];
    }
    cout &lt;&lt; endl;
    
    return 0;
}

```
