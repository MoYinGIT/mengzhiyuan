## 📝 代码实现

### C++版本

```cpp
/*
 * 题目28：A1170 计算2的N次方
 */

#include &lt;iostream&gt;
using namespace std;

int main() {
    int n;
    cin &gt;&gt; n;
    
    int result[1000] = {0};
    result[0] = 1;
    int len = 1;
    
    for (int i = 0; i &lt; n; i++) {
        int carry = 0;
        for (int j = 0; j &lt; len; j++) {
            int product = result[j] * 2 + carry;
            result[j] = product % 10;
            carry = product / 10;
        }
        while (carry) {
            result[len++] = carry % 10;
            carry /= 10;
        }
    }
    
    for (int i = len - 1; i &gt;= 0; i--) {
        cout &lt;&lt; result[i];
    }
    cout &lt;&lt; endl;
    
    return 0;
}

```
