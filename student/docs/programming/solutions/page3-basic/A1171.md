## 📝 代码实现

### C++版本

```cpp
/*
 * 题目29：A1171 大整数的因子
 */

#include &lt;iostream&gt;
#include &lt;cmath&gt;
#include &lt;string&gt;
using namespace std;

long long str_to_ll(string s) {
    long long res = 0;
    for (char c : s) {
        res = res * 10 + (c - '0');
    }
    return res;
}

int main() {
    string s;
    cin &gt;&gt; s;
    
    long long n = str_to_ll(s);
    
    int count = 0;
    long long i;
    for (i = 1; i * i &lt; n; i++) {
        if (n % i == 0) {
            count += 2;
        }
    }
    if (i * i == n) count++;
    
    cout &lt;&lt; count &lt;&lt; endl;
    
    return 0;
}

```
