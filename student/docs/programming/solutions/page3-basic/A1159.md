## 📝 代码实现

### C语言版本

```c
/*
 * 题目17：A1159 斐波那契数列
 * 
 * F(1)=1, F(2)=1, F(n)=F(n-1)+F(n-2)
 * 求第n项
 */

#include &lt;stdio.h&gt;

int main() {
    int n;
    scanf("%d", &n);
    
    if (n == 1 || n == 2) {
        printf("1\n");
        return 0;
    }
    
    long long a = 1, b = 1, c;
    for (int i = 3; i &lt;= n; i++) {
        c = a + b;
        a = b;
        b = c;
    }
    
    printf("%lld\n", b);
    
    return 0;
}

```
