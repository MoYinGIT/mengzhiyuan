## 📝 代码实现

### C++版本

```cpp
/*
 * 题目4：A1123 图像相似度
 * 
 * 计算两幅图像的相似度（像素差值绝对值的平均值）
 */

#include &lt;iostream&gt;
#include &lt;cmath&gt;
using namespace std;

int main() {
    int m, n;
    int a[100][100], b[100][100];
    
    cin &gt;&gt; m &gt;&gt; n;
    
    for (int i = 0; i &lt; m; i++) {
        for (int j = 0; j &lt; n; j++) {
            cin &gt;&gt; a[i][j];
        }
    }
    
    for (int i = 0; i &lt; m; i++) {
        for (int j = 0; j &lt; n; j++) {
            cin &gt;&gt; b[i][j];
        }
    }
    
    long long sum = 0;
    for (int i = 0; i &lt; m; i++) {
        for (int j = 0; j &lt; n; j++) {
            sum += abs(a[i][j] - b[i][j]);
        }
    }
    
    double avg = (double)sum / (m * n);
    printf("%.2f\n", avg);
    
    return 0;
}

```
