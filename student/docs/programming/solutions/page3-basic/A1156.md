## 📝 代码实现

### C++版本

```cpp
/*
 * 题目14：A1156 求π的值
 * 
 * 使用公式：π/4 = 1 - 1/3 + 1/5 - 1/7 + ...
 */

#include &lt;iostream&gt;
#include &lt;cmath&gt;
using namespace std;

int main() {
    double eps;
    cin &gt;&gt; eps;
    
    double pi = 0.0;
    double term;
    int i = 0;
    
    do {
        term = 1.0 / (2 * i + 1);
        if (i % 2 == 0) {
            pi += term;
        } else {
            pi -= term;
        }
        i++;
    } while (fabs(term) &gt;= eps);
    
    pi *= 4;
    printf("%.6f\n", pi);
    
    return 0;
}

```
