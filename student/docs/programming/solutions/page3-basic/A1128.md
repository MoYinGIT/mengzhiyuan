## 📝 代码实现

### C语言版本

```c
/*
 * 题目6：A1128 图像模糊处理
 * 
 * 每个像素的新值 = 上下左右四个相邻像素（含自己）的平均值（向下取整）
 * 
 * 注意：边界上的点只考虑存在的邻居
 */

#include &lt;stdio.h&gt;

int main() {
    int n, m;
    int a[100][100], b[100][100];
    
    scanf("%d %d", &n, &m);
    
    for (int i = 0; i &lt; n; i++) {
        for (int j = 0; j &lt; m; j++) {
            scanf("%d", &a[i][j]);
        }
    }
    
    for (int i = 0; i &lt; n; i++) {
        for (int j = 0; j &lt; m; j++) {
            int sum = a[i][j];
            int count = 1;
            
            // 上
            if (i &gt; 0) {
                sum += a[i-1][j];
                count++;
            }
            // 下
            if (i &lt; n - 1) {
                sum += a[i+1][j];
                count++;
            }
            // 左
            if (j &gt; 0) {
                sum += a[i][j-1];
                count++;
            }
            // 右
            if (j &lt; m - 1) {
                sum += a[i][j+1];
                count++;
            }
            
            b[i][j] = sum / count;
        }
    }
    
    for (int i = 0; i &lt; n; i++) {
        for (int j = 0; j &lt; m; j++) {
            if (j &gt; 0) printf(" ");
            printf("%d", b[i][j]);
        }
        printf("\n");
    }
    
    return 0;
}

```
