## 📝 代码实现

### C++版本

```cpp
/*
 * 题目19：A1161 进制转换
 * 
 * 将10进制数转换为m进制（m &lt;= 16）
 */

#include &lt;iostream&gt;
#include &lt;string&gt;
using namespace std;

int main() {
    int n, m;
    cin &gt;&gt; n &gt;&gt; m;
    
    if (n == 0) {
        cout &lt;&lt; 0 &lt;&lt; endl;
        return 0;
    }
    
    char digits[] = "0123456789ABCDEF";
    char result[100];
    int idx = 0;
    
    int negative = 0;
    if (n &lt; 0) {
        negative = 1;
        n = -n;
    }
    
    while (n &gt; 0) {
        result[idx++] = digits[n % m];
        n /= m;
    }
    
    if (negative) {
        cout &lt;&lt; "-";
    }
    
    for (int i = idx - 1; i &gt;= 0; i--) {
        cout &lt;&lt; result[i];
    }
    cout &lt;&lt; endl;
    
    return 0;
}

```
