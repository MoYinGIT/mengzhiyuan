/**
 * A1253: 迷宫问题
 * BFS + 路径记录，需要输出最短路径的具体走法
 * 
 * 需要记录路径，输出路径坐标
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin &gt;&gt; n &gt;&gt; m;
    
    vector&lt;vector&lt;int&gt;&gt; maze(n, vector&lt;int&gt;(m));
    for (int i = 0; i &lt; n; i++) {
        for (int j = 0; j &lt; m; j++) {
            cin &gt;&gt; maze[i][j];
        }
    }
    
    // BFS
    vector&lt;vector&lt;bool&gt;&gt; vis(n, vector&lt;bool&gt;(m, false));
    vector&lt;vector&lt;pair&lt;int, int&gt;&gt;&gt; pre(n, vector&lt;pair&lt;int, int&gt;&gt;(m, {-1, -1}));
    queue&lt;pair&lt;int, int&gt;&gt; q;
    
    // 假设起点是(0,0)，终点是(n-1,m-1)
    // 0表示通路，1表示障碍
    q.push({0, 0});
    vis[0][0] = true;
    
    int dx[] = {-1, 1, 0, 0};
    int dy[] = {0, 0, -1, 1};
    
    bool found = false;
    
    while (!q.empty() && !found) {
        auto [x, y] = q.front(); q.pop();
        
        if (x == n - 1 && y == m - 1) {
            found = true;
            break;
        }
        
        for (int i = 0; i &lt; 4; i++) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            
            if (nx &gt;= 0 && nx &lt; n && ny &gt;= 0 && ny &lt; m && 
                !vis[nx][ny] && maze[nx][ny] == 0) {
                vis[nx][ny] = true;
                pre[nx][ny] = {x, y};
                q.push({nx, ny});
            }
        }
    }
    
    // 还原路径
    vector&lt;pair&lt;int, int&gt;&gt; path;
    int x = n - 1, y = m - 1;
    while (x != -1) {
        path.push_back({x, y});
        auto [px, py] = pre[x][y];
        if (x == 0 && y == 0) break;
        x = px; y = py;
    }
    
    // 输出路径（从起点到终点）
    for (int i = path.size() - 1; i &gt;= 0; i--) {
        cout &lt;&lt; "(" &lt;&lt; path[i].first &lt;&lt; "," &lt;&lt; path[i].second &lt;&lt; ")";
        if (i &gt; 0) cout &lt;&lt; " ";
    }
    cout &lt;&lt; endl;
    
    return 0;
}
