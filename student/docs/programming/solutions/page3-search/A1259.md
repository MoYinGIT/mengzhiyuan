/**
 * A1259: 马走日
 * 类似Knight Moves，但求路径数量或具体路径
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int n, m, sx, sy, ex, ey;
int ans = 0;
bool vis[15][15];

// 马的8种走法
int dx[] = {-2, -2, -1, -1, 1, 1, 2, 2};
int dy[] = {-1, 1, -2, 2, -2, 2, -1, 1};

void dfs(int x, int y, int step) {
    if (x == ex && y == ey) {
        if (step == n * m) {
            ans++;
        }
        return;
    }
    
    for (int i = 0; i &lt; 8; i++) {
        int nx = x + dx[i];
        int ny = y + dy[i];
        
        if (nx &gt;= 0 && nx &lt; n && ny &gt;= 0 && ny &lt; m && !vis[nx][ny]) {
            vis[nx][ny] = true;
            dfs(nx, ny, step + 1);
            vis[nx][ny] = false;
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int t;
    cin &gt;&gt; t;
    
    while (t--) {
        cin &gt;&gt; n &gt;&gt; m &gt;&gt; sx &gt;&gt; sy;
        ex = 0; ey = 0;  // 假设终点是(0,0)或需要其他处理
        
        // 根据题目要求，可能是求访问所有格子的路径数
        // 或者求到某个终点的路径数
        // 这里假设是求马的遍历路径数
        
        ans = 0;
        memset(vis, false, sizeof(vis));
        vis[sx][sy] = true;
        dfs(sx, sy, 1);
        
        cout &lt;&lt; ans &lt;&lt; endl;
    }
    
    return 0;
}
