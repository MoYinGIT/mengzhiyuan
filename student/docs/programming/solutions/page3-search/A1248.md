## 📝 代码实现

### C++版本

```cpp
/**
 * A1248: The Castle
 * 城堡问题 - 统计房间数量和最大房间面积
 * 用位运算表示墙壁，Flood Fill求连通块
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin &gt;&gt; n &gt;&gt; m;
    
    vector&lt;vector&lt;int&gt;&gt; castle(n, vector&lt;int&gt;(m));
    for (int i = 0; i &lt; n; i++) {
        for (int j = 0; j &lt; m; j++) {
            cin &gt;&gt; castle[i][j];
        }
    }
    
    vector&lt;vector&lt;bool&gt;&gt; vis(n, vector&lt;bool&gt;(m, false));
    int roomCount = 0;
    int maxArea = 0;
    
    // 四个方向：西(1)、北(2)、东(4)、南(8)
    // 对应dx,dy: 西(0,-1)、北(-1,0)、东(0,1)、南(1,0)
    int dx[] = {0, -1, 0, 1};
    int dy[] = {-1, 0, 1, 0};
    int wallBit[] = {1, 2, 4, 8};  // 西、北、东、南
    
    for (int i = 0; i &lt; n; i++) {
        for (int j = 0; j &lt; m; j++) {
            if (!vis[i][j]) {
                // 新房间
                roomCount++;
                int area = 0;
                queue&lt;pair&lt;int, int&gt;&gt; q;
                q.push({i, j});
                vis[i][j] = true;
                
                while (!q.empty()) {
                    auto [x, y] = q.front(); q.pop();
                    area++;
                    
                    for (int k = 0; k &lt; 4; k++) {
                        // 检查这个方向是否有墙
                        if (castle[x][y] & wallBit[k]) continue;  // 有墙
                        
                        int nx = x + dx[k];
                        int ny = y + dy[k];
                        
                        if (nx &gt;= 0 && nx &lt; n && ny &gt;= 0 && ny &lt; m &&
                            !vis[nx][ny]) {
                            vis[nx][ny] = true;
                            q.push({nx, ny});
                        }
                    }
                }
                
                maxArea = max(maxArea, area);
            }
        }
    }
    
    cout &lt;&lt; roomCount &lt;&lt; endl;
    cout &lt;&lt; maxArea &lt;&lt; endl;
    
    return 0;
}

```
