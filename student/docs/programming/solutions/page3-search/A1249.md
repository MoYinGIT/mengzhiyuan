```cpp
/**
 * A1249: 仙岛求药
 * BFS求最短路径，有障碍物和传送点
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    while (true) {
        int n, m;
        cin &gt;&gt; n &gt;&gt; m;
        if (n == 0 && m == 0) break;
        
        vector&lt;string&gt; grid(n);
        for (int i = 0; i &lt; n; i++) {
            cin &gt;&gt; grid[i];
        }
        
        // 找起点@和终点*
        int sx = -1, sy = -1, ex = -1, ey = -1;
        for (int i = 0; i &lt; n; i++) {
            for (int j = 0; j &lt; m; j++) {
                if (grid[i][j] == '@') {
                    sx = i; sy = j;
                } else if (grid[i][j] == '*') {
                    ex = i; ey = j;
                }
            }
        }
        
        // BFS
        vector&lt;vector&lt;int&gt;&gt; dist(n, vector&lt;int&gt;(m, -1));
        queue&lt;pair&lt;int, int&gt;&gt; q;
        
        dist[sx][sy] = 0;
        q.push({sx, sy});
        
        int dx[] = {-1, 1, 0, 0};
        int dy[] = {0, 0, -1, 1};
        
        while (!q.empty()) {
            auto [x, y] = q.front(); q.pop();
            
            if (x == ex && y == ey) {
                cout &lt;&lt; dist[x][y] &lt;&lt; endl;
                break;
            }
            
            for (int i = 0; i &lt; 4; i++) {
                int nx = x + dx[i];
                int ny = y + dy[i];
                
                if (nx &gt;= 0 && nx &lt; n && ny &gt;= 0 && ny &lt; m &&
                    dist[nx][ny] == -1 && grid[nx][ny] != '#') {
                    dist[nx][ny] = dist[x][y] + 1;
                    q.push({nx, ny});
                }
            }
        }
        
        if (dist[ex][ey] == -1) {
            cout &lt;&lt; -1 &lt;&lt; endl;
        }
    }
    
    return 0;
}

```
