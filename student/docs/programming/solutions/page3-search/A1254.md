/**
 * A1254: 细胞计数
 * Flood Fill统计连通块
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin &gt;&gt; n &gt;&gt; m;
    
    vector&lt;string&gt; grid(n);
    for (int i = 0; i &lt; n; i++) {
        cin &gt;&gt; grid[i];
    }
    
    vector&lt;vector&lt;bool&gt;&gt; vis(n, vector&lt;bool&gt;(m, false));
    int count = 0;
    
    // 四连通或八连通，这里用四连通
    int dx[] = {-1, 1, 0, 0};
    int dy[] = {0, 0, -1, 1};
    
    for (int i = 0; i &lt; n; i++) {
        for (int j = 0; j &lt; m; j++) {
            if (grid[i][j] != '0' && !vis[i][j]) {
                count++;
                queue&lt;pair&lt;int, int&gt;&gt; q;
                q.push({i, j});
                vis[i][j] = true;
                
                while (!q.empty()) {
                    auto [x, y] = q.front(); q.pop();
                    
                    for (int k = 0; k &lt; 4; k++) {
                        int nx = x + dx[k];
                        int ny = y + dy[k];
                        
                        if (nx &gt;= 0 && nx &lt; n && ny &gt;= 0 && ny &lt; m &&
                            grid[nx][ny] != '0' && !vis[nx][ny]) {
                            vis[nx][ny] = true;
                            q.push({nx, ny});
                        }
                    }
                }
            }
        }
    }
    
    cout &lt;&lt; count &lt;&lt; endl;
    
    return 0;
}
