```cpp
/**
 * A1251: 抓住那头牛
 * 一维BFS，三种移动方式：-1, +1, *2
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

const int MAX = 100005;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, k;
    cin &gt;&gt; n &gt;&gt; k;
    
    if (n &gt;= k) {
        cout &lt;&lt; n - k &lt;&lt; endl;
        return 0;
    }
    
    vector&lt;int&gt; dist(MAX, -1);
    queue&lt;int&gt; q;
    
    dist[n] = 0;
    q.push(n);
    
    while (!q.empty()) {
        int x = q.front(); q.pop();
        
        if (x == k) {
            cout &lt;&lt; dist[x] &lt;&lt; endl;
            return 0;
        }
        
        // 三种移动
        int nextPos[3] = {x - 1, x + 1, x * 2};
        for (int nx : nextPos) {
            if (nx &gt;= 0 && nx &lt; MAX && dist[nx] == -1) {
                dist[nx] = dist[x] + 1;
                q.push(nx);
            }
        }
    }
    
    return 0;
}

```
