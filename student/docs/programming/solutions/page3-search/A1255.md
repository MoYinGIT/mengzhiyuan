/**
 * A1255: Knight Moves
 * 国际象棋马走日，求最少步数
 */

#include &lt;bits/stdc++.h&gt;
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int t;
    cin &gt;&gt; t;
    
    // 马的8种走法
    int dx[] = {-2, -2, -1, -1, 1, 1, 2, 2};
    int dy[] = {-1, 1, -2, 2, -2, 2, -1, 1};
    
    while (t--) {
        int l;
        cin &gt;&gt; l;
        
        int sx, sy, ex, ey;
        cin &gt;&gt; sx &gt;&gt; sy;
        cin &gt;&gt; ex &gt;&gt; ey;
        
        if (sx == ex && sy == ey) {
            cout &lt;&lt; 0 &lt;&lt; endl;
            continue;
        }
        
        vector&lt;vector&lt;int&gt;&gt; dist(l, vector&lt;int&gt;(l, -1));
        queue&lt;pair&lt;int, int&gt;&gt; q;
        
        dist[sx][sy] = 0;
        q.push({sx, sy});
        
        bool found = false;
        
        while (!q.empty() && !found) {
            auto [x, y] = q.front(); q.pop();
            
            for (int i = 0; i &lt; 8; i++) {
                int nx = x + dx[i];
                int ny = y + dy[i];
                
                if (nx &gt;= 0 && nx &lt; l && ny &gt;= 0 && ny &lt; l &&
                    dist[nx][ny] == -1) {
                    dist[nx][ny] = dist[x][y] + 1;
                    
                    if (nx == ex && ny == ey) {
                        cout &lt;&lt; dist[nx][ny] &lt;&lt; endl;
                        found = true;
                        break;
                    }
                    
                    q.push({nx, ny});
                }
            }
        }
    }
    
    return 0;
}
