## 📚 模块一：数组进阶

### 1.1 数组回顾与进阶

**数组声明与初始化**：

```cpp
#include <iostream>
using namespace std;

int main() {
    // 方式1：完全初始化
    int arr1[5] = {1, 2, 3, 4, 5};
    
    // 方式2：部分初始化（剩余为0）
    int arr2[5] = {1, 2};  // {1, 2, 0, 0, 0}
    
    // 方式3：自动计算长度
    int arr3[] = {1, 2, 3, 4, 5};  // 长度自动为5
    
    // 方式4：全部初始化为0
    int arr4[100] = {0};  // 所有元素都是0
    
    return 0;
}
```

**数组越界警告**：

```cpp
int arr[5] = {1, 2, 3, 4, 5};

// ❌ 错误！数组索引从0到4，没有arr[5]

```

### 1.2 数组常用操作

**查找最大值/最小值**：

```cpp
#include <iostream>
using namespace std;

int main() {
    int arr[10] = {23, 56, 12, 89, 45, 67, 34, 91, 8, 50};
    int n = 10;
    
    // 假设第一个元素是最大值
    int max_val = arr[0];
    int min_val = arr[0];
    
    for (int i = 1; i < n; i++) {
        if (arr[i] > max_val) {
            max_val = arr[i];
        }
        if (arr[i] < min_val) {
            min_val = arr[i];
        }
    }
    
    cout << "最大值: " << max_val << endl;  // 91
    cout << "最小值: " << min_val << endl;  // 8
    
    return 0;
}
```

**数组求和与平均值**：

```cpp
int sum = 0;
for (int i = 0; i < n; i++) {

}

```

### 1.3 数组排序（冒泡排序）

**原理**：相邻元素两两比较，大的往后冒泡。

```cpp
#include <iostream>
using namespace std;

int main() {
    int arr[5] = {64, 34, 25, 12, 22};
    int n = 5;
    
    // 冒泡排序
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                // 交换
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
    
    // 输出排序后的数组
    cout << "排序后: ";
    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
    
    return 0;
}
```

**排序过程可视化**：
```
初始: [64, 34, 25, 12, 22]

第1轮: [34, 25, 12, 22, 64]  // 64冒泡到最后
第2轮: [25, 12, 22, 34, 64]  // 34冒泡到倒数第二
第3轮: [12, 22, 25, 34, 64]  // 25归位
第4轮: [12, 22, 25, 34, 64]  // 完成
```

## 📚 模块二：字符串

### 2.1 string 类型

```cpp
#include <iostream>
#include <string>  // 字符串操作头文件
using namespace std;

int main() {
    // 声明字符串
    string s1 = "Hello";
    string s2 = "World";
    
    // 字符串连接
    string s3 = s1 + " " + s2;  // "Hello World"
    cout << s3 << endl;
    
    // 获取长度
    cout << "长度: " << s3.length() << endl;  // 11
    
    // 访问字符
    cout << "第1个字符: " << s3[0] << endl;     // H
    cout << "最后一个字符: " << s3[s3.length()-1] << endl;  // d
    
    return 0;
}
```

### 2.2 常用字符串操作

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string s = "Hello C++";
    
    // 1. 获取子串
    string sub = s.substr(6, 3);  // 从索引6开始，取3个字符
    cout << sub << endl;  // "C++"
    
    // 2. 查找子串
    size_t pos = s.find("C++");
    if (pos != string::npos) {
        cout << "找到C++，位置: " << pos << endl;
    }
    
    // 3. 替换子串
    s.replace(6, 3, "World");  // 将"C++"替换为"World"
    cout << s << endl;  // "Hello World"
    
    // 4. 插入字符
    s.insert(5, " Beautiful");  // 在索引5处插入
    cout << s << endl;  // "Hello Beautiful World"
    
    // 5. 删除子串
    s.erase(5, 10);  // 从索引5开始删除10个字符
    cout << s << endl;  // "Hello World"
    
    // 6. 比较字符串
    string a = "abc";
    string b = "def";
    if (a < b) {
        cout << "a 小于 b" << endl;  // 字典序比较
    }
    
    return 0;
}
```

### 2.3 字符串遍历与修改

```cpp
#include <iostream>
#include <cctype>  // 字符处理函数
using namespace std;

int main() {
    string s = "Hello World";
    
    // 遍历字符串
    for (int i = 0; i < s.length(); i++) {
        cout << s[i] << " ";
    }
    cout << endl;
    
    // 转大写
    for (int i = 0; i < s.length(); i++) {
        s[i] = toupper(s[i]);
    }
    cout << s << endl;  // "HELLO WORLD"
    
    // 转小写
    for (int i = 0; i < s.length(); i++) {
        s[i] = tolower(s[i]);
    }
    cout << s << endl;  // "hello world"
    
    // 统计字母个数
    string text = "Hello 123";
    int letter_count = 0;
    for (int i = 0; i < text.length(); i++) {
        if (isalpha(text[i])) {  // 是否是字母
            letter_count++;
        }
    }
    cout << "字母个数: " << letter_count << endl;  // 5
    
    return 0;
}
```

## 📚 模块三：函数

### 3.1 函数基础

**为什么用函数？**
- 代码复用（写一次，用多次）
- 模块化（分解复杂问题）
- 可读性（命名清晰的功能块）

**函数定义**：

```cpp
#include <iostream>
using namespace std;

// 函数定义
// 返回值类型 函数名(参数列表) {
//     函数体
//     return 返回值;
// }

// 无参数、无返回值的函数
void sayHello() {
    cout << "Hello!" << endl;
}

// 有参数、无返回值的函数
void greet(string name) {
    cout << "Hello, " << name << "!" << endl;
}

// 有参数、有返回值的函数
int add(int a, int b) {
    return a + b;
}

int main() {
    // 调用函数
    sayHello();           // 输出: Hello!
    greet("小明");         // 输出: Hello, 小明!
    
    int result = add(3, 5);
    cout << result << endl;  // 输出: 8
    
    return 0;
}
```

### 3.2 函数参数传递

**值传递**（默认）：

```cpp
void swap(int a, int b) {
    int temp = a;
    a = b;
    b = temp;
    // 这里的交换只在函数内部有效
}

int main() {
    int x = 3, y = 5;
    swap(x, y);
    cout << x << " " << y << endl;  // 还是 3 5
    return 0;
}
```

**引用传递**（L3学习）：

```cpp
void swap(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

int main() {
    int x = 3, y = 5;
    swap(x, y);
    cout << x << " " << y << endl;  // 变成 5 3
    return 0;
}
```

### 3.3 函数重载（了解）

同一个函数名，不同参数：

```cpp
int add(int a, int b) {
    return a + b;
}

double add(double a, double b) {
    return a + b;
}

int main() {
    cout << add(2, 3) << endl;       // 调用 int 版本
    cout << add(2.5, 3.5) << endl;  // 调用 double 版本
    return 0;
}
```

### 3.4 递归函数（初步）

函数调用自己：

```cpp
// 计算阶乘
// n! = n * (n-1) * (n-2) * ... * 1
// 5! = 5 * 4 * 3 * 2 * 1 = 120

int factorial(int n) {
    if (n <= 1) {
        return 1;  // 基准情况
    }
    return n * factorial(n - 1);  // 递归调用
}

int main() {
    cout << factorial(5) << endl;  // 120
    return 0;
}
```

**递归的关键**：
1. 基准情况（终止条件）
2. 递归步骤（向基准靠近）

## 📚 模块四：作用域与变量

### 4.1 局部变量与全局变量

```cpp
#include <iostream>
using namespace std;

// 全局变量（所有函数都能访问）
int global_var = 100;

void func1() {
    // 局部变量（只在func1中有效）
    int local_var = 10;
    cout << "func1: " << local_var << endl;
    cout << "func1: " << global_var << endl;
}

void func2() {
    // 这里不能访问 local_var
    // cout << local_var;  // 错误！
    
    cout << "func2: " << global_var << endl;
}

int main() {
    func1();
    func2();
    return 0;
}
```

**建议**：尽量少用全局变量，多用参数传递。

### 4.2 变量的生命周期

```cpp
void demo() {
    int a = 0;        // 每次调用都重新创建
    static int b = 0; // 只初始化一次，值会保留
    
    a++;
    b++;
    
    cout << "a=" << a << ", b=" << b << endl;
}

int main() {
    demo();  // a=1, b=1
    demo();  // a=1, b=2
    demo();  // a=1, b=3
    return 0;
}
```

## 🎮 综合项目

### 项目一：学生成绩管理系统

```cpp
#include <iostream>
#include <iomanip>
using namespace std;

const int MAX_STUDENTS = 100;

// 显示菜单
void showMenu() {
    cout << "\n===== 学生成绩管理系统 =====" << endl;
    cout << "1. 输入成绩" << endl;
    cout << "2. 显示成绩" << endl;
    cout << "3. 计算平均分" << endl;
    cout << "4. 查找最高分和最低分" << endl;
    cout << "5. 成绩排序" << endl;
    cout << "0. 退出" << endl;
    cout << "请选择: ";
}

// 输入成绩
void inputScores(int scores[], int &n) {
    cout << "请输入学生人数: ";
    cin >> n;
    
    for (int i = 0; i < n; i++) {
        cout << "请输入第" << i + 1 << "个学生的成绩: ";
        cin >> scores[i];
    }
}

// 显示成绩
void displayScores(int scores[], int n) {
    cout << "\n成绩列表:" << endl;
    for (int i = 0; i < n; i++) {
        cout << "学生" << i + 1 << ": " << scores[i] << endl;
    }
}

// 计算平均分
double getAverage(int scores[], int n) {
    int sum = 0;
    for (int i = 0; i < n; i++) {
        sum += scores[i];
    }
    return (double)sum / n;
}

// 查找最高分和最低分
void findMinMax(int scores[], int n, int &min_val, int &max_val) {
    min_val = max_val = scores[0];
    for (int i = 1; i < n; i++) {
        if (scores[i] < min_val) min_val = scores[i];
        if (scores[i] > max_val) max_val = scores[i];
    }
}

// 冒泡排序
void bubbleSort(int scores[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1 - i; j++) {
            if (scores[j] > scores[j + 1]) {
                int temp = scores[j];
                scores[j] = scores[j + 1];
                scores[j + 1] = temp;
            }
        }
    }
}

int main() {
    int scores[MAX_STUDENTS];
    int n = 0;
    int choice;
    
    do {
        showMenu();
        cin >> choice;
        
        switch (choice) {
            case 1:
                inputScores(scores, n);
                break;
            case 2:
                displayScores(scores, n);
                break;
            case 3:
                cout << "平均分: " << fixed << setprecision(2) 
                     << getAverage(scores, n) << endl;
                break;
            case 4:
                int min_val, max_val;
                findMinMax(scores, n, min_val, max_val);
                cout << "最低分: " << min_val << endl;
                cout << "最高分: " << max_val << endl;
                break;
            case 5:
                bubbleSort(scores, n);
                cout << "已排序!" << endl;
                displayScores(scores, n);
                break;
            case 0:
                cout << "再见!" << endl;
                break;
            default:
                cout << "无效选择!" << endl;
        }
    } while (choice != 0);
    
    return 0;
}
```

## 🎯 三经典智慧

### 《易经》· 模块化思维

函数就是「模块化」：
- 大化小，小化了
- 复杂问题分解为简单函数
- 每个函数只做一件事

### 《道德经》· 复用之道

函数体现了「无为而无不为」：
- 写一次，用无数次
- 不重复造轮子
- 简洁高效的代码

### 《论语》· 教学相长

编写函数的过程：
- 明确输入（参数）
- 明确输出（返回值）
- 清晰的逻辑（函数体）

这就像准备一堂课，目标明确才能教得好。
