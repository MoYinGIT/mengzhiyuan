### 1.1 并查集（Union-Find）

**用途**：维护集合的合并与查询

**基本操作**：
- `find(x)`：查找 x 所在集合的代表（根）
- `union(x, y)`：合并 x 和 y 所在的集合

**代码实现（路径压缩 + 按秩合并）**：

```cpp
#include <iostream>
using namespace std;

const int MAXN = 100005;

int parent[MAXN];  // parent[i] 表示 i 的父节点
int rank_[MAXN];   // rank[i] 表示以 i 为根的树的高度

// 初始化
void init(int n) {
    for (int i = 0; i <= n; i++) {
        parent[i] = i;
        rank_[i] = 0;
    }
}

// 查找（带路径压缩）
int find(int x) {
    if (parent[x] != x) {
        parent[x] = find(parent[x]);  // 路径压缩
    }
    return parent[x];
}

// 合并（按秩合并）
void union_set(int x, int y) {
    int px = find(x), py = find(y);
    if (px == py) return;
    
    if (rank_[px] < rank_[py]) swap(px, py);
    parent[py] = px;
    if (rank_[px] == rank_[py]) rank_[px]++;
}

// 判断是否在同一集合
bool same(int x, int y) {
    return find(x) == find(y);
}
```

**应用场景**：
- 连通分量判断
- Kruskal 最小生成树
- 离线查询

### 1.2 树状数组（Binary Indexed Tree）

**用途**：高效维护前缀和，支持单点修改和区间查询

**核心思想**：利用二进制性质，将区间分解为 O(log n) 个子区间

```cpp
#include <iostream>
using namespace std;

const int MAXN = 100005;

int tree[MAXN];
int n;

// lowbit: 获取最低位的1
int lowbit(int x) {
    return x & (-x);
}

// 在位置 i 增加 val
void add(int i, int val) {
    while (i <= n) {
        tree[i] += val;
        i += lowbit(i);
    }
}

// 查询前缀和 [1, i]
int query(int i) {
    int sum = 0;
    while (i > 0) {
        sum += tree[i];
        i -= lowbit(i);
    }
    return sum;
}

// 查询区间和 [l, r]
int range_query(int l, int r) {
    return query(r) - query(l - 1);
}
```

**应用场景**：
- 动态前缀和
- 逆序对统计
- 区间修改单点查询（差分）

### 1.3 线段树（Segment Tree）

**用途**：维护区间信息，支持区间修改和区间查询

**基本结构**：
```
        [1, 8]
       /      \
    [1,4]    [5,8]
    /   \     /   \
  [1,2][3,4][5,6][7,8]
```

**代码实现（区间求和）**：

```cpp
#include <iostream>
#include <vector>
using namespace std;

const int MAXN = 100005;

struct SegmentTree {
    long long sum[MAXN * 4];   // 区间和
    long long lazy[MAXN * 4];  // 懒标记
    int n;
    
    // 建树
    void build(int node, int l, int r, int arr[]) {
        if (l == r) {
            sum[node] = arr[l];
            return;
        }
        int mid = (l + r) / 2;
        build(node * 2, l, mid, arr);
        build(node * 2 + 1, mid + 1, r, arr);
        sum[node] = sum[node * 2] + sum[node * 2 + 1];
    }
    
    // 下传懒标记
    void push_down(int node, int l, int r) {
        if (lazy[node] != 0) {
            int mid = (l + r) / 2;
            sum[node * 2] += lazy[node] * (mid - l + 1);
            sum[node * 2 + 1] += lazy[node] * (r - mid);
            lazy[node * 2] += lazy[node];
            lazy[node * 2 + 1] += lazy[node];
            lazy[node] = 0;
        }
    }
    
    // 区间修改 [L, R] 加上 val
    void update(int node, int l, int r, int L, int R, long long val) {
        if (L <= l && r <= R) {
            sum[node] += val * (r - l + 1);
            lazy[node] += val;
            return;
        }
        push_down(node, l, r);
        int mid = (l + r) / 2;
        if (L <= mid) update(node * 2, l, mid, L, R, val);
        if (R > mid) update(node * 2 + 1, mid + 1, r, L, R, val);
        sum[node] = sum[node * 2] + sum[node * 2 + 1];
    }
    
    // 区间查询 [L, R]
    long long query(int node, int l, int r, int L, int R) {
        if (L <= l && r <= R) return sum[node];
        push_down(node, l, r);
        int mid = (l + r) / 2;
        long long res = 0;
        if (L <= mid) res += query(node * 2, l, mid, L, R);
        if (R > mid) res += query(node * 2 + 1, mid + 1, r, L, R);
        return res;
    }
};
```

**应用场景**：
- 区间求和/最值
- 区间修改
- 区间覆盖

### 1.4 Trie 树（字典树）

**用途**：高效存储和检索字符串

```cpp
#include <iostream>
#include <string>
using namespace std;

const int MAXN = 100005;
const int CHARSET = 26;  // 假设只有小写字母

struct Trie {
    int trie[MAXN][CHARSET];
    int count[MAXN];  // 以该节点结尾的单词数
    int node_count;
    
    Trie() {
        node_count = 0;
        memset(trie, 0, sizeof(trie));
        memset(count, 0, sizeof(count));
    }
    
    // 插入单词
    void insert(const string& word) {
        int node = 0;
        for (char c : word) {
            int idx = c - 'a';
            if (trie[node][idx] == 0) {
                trie[node][idx] = ++node_count;
            }
            node = trie[node][idx];
        }
        count[node]++;
    }
    
    // 查询单词出现次数
    int search(const string& word) {
        int node = 0;
        for (char c : word) {
            int idx = c - 'a';
            if (trie[node][idx] == 0) return 0;
            node = trie[node][idx];
        }
        return count[node];
    }
    
    // 查询前缀出现次数
    int startsWith(const string& prefix) {
        int node = 0;
        for (char c : prefix) {
            int idx = c - 'a';
            if (trie[node][idx] == 0) return 0;
            node = trie[node][idx];
        }
        return count[node];  // 可以修改返回所有子树单词数
    }
};
```

## 📚 模块二：网络流

### 2.1 最大流问题

**问题**：在有向图中，从源点 s 到汇点 t，每条边有容量限制，求最大流量

**Ford-Fulkerson 方法（ Edmonds-Karp 算法）**：

```cpp
#include <iostream>
#include <queue>
#include <cstring>
using namespace std;

const int MAXN = 205;
const int INF = 0x3f3f3f3f;

int capacity[MAXN][MAXN];  // 残量网络
int flow[MAXN][MAXN];      // 流量
int parent[MAXN];
int n, m, s, t;

// BFS 找增广路
bool bfs() {
    memset(parent, -1, sizeof(parent));
    queue<int> q;
    q.push(s);
    parent[s] = s;
    
    while (!q.empty() && parent[t] == -1) {
        int u = q.front(); q.pop();
        for (int v = 1; v <= n; v++) {
            if (parent[v] == -1 && capacity[u][v] > flow[u][v]) {
                parent[v] = u;
                q.push(v);
            }
        }
    }
    return parent[t] != -1;
}

// Edmonds-Karp 算法
int max_flow() {
    int maxFlow = 0;
    while (bfs()) {
        // 找到增广路上的最小残量
        int minResidual = INF;
        for (int v = t; v != s; v = parent[v]) {
            int u = parent[v];
            minResidual = min(minResidual, capacity[u][v] - flow[u][v]);
        }
        
        // 更新流量
        for (int v = t; v != s; v = parent[v]) {
            int u = parent[v];
            flow[u][v] += minResidual;
            flow[v][u] -= minResidual;
        }
        
        maxFlow += minResidual;
    }
    return maxFlow;
}
```

### 2.2 Dinic 算法

**更高效的算法，时间复杂度 O(V²E)**

```cpp
#include <iostream>
#include <queue>
#include <cstring>
using namespace std;

const int MAXN = 10005;
const int INF = 0x3f3f3f3f;

struct Edge {
    int to, capacity, flow, reverse;
};

vector<Edge> graph[MAXN];
int level[MAXN], iter[MAXN];

// 添加边
void add_edge(int from, int to, int capacity) {
    graph[from].push_back({to, capacity, 0, (int)graph[to].size()});
    graph[to].push_back({from, 0, 0, (int)graph[from].size() - 1});
}

// BFS 构建分层图
bool bfs(int s, int t) {
    memset(level, -1, sizeof(level));
    queue<int> q;
    level[s] = 0;
    q.push(s);
    while (!q.empty()) {
        int v = q.front(); q.pop();
        for (const Edge& e : graph[v]) {
            if (level[e.to] < 0 && e.flow < e.capacity) {
                level[e.to] = level[v] + 1;
                q.push(e.to);
            }
        }
    }
    return level[t] >= 0;
}

// DFS 找增广路
int dfs(int v, int t, int f) {
    if (v == t) return f;
    for (int& i = iter[v]; i < graph[v].size(); i++) {
        Edge& e = graph[v][i];
        if (e.flow < e.capacity && level[v] < level[e.to]) {
            int d = dfs(e.to, t, min(f, e.capacity - e.flow));
            if (d > 0) {
                e.flow += d;
                graph[e.to][e.reverse].flow -= d;
                return d;
            }
        }
    }
    return 0;
}

// Dinic 算法
int dinic(int s, int t) {
    int flow = 0;
    while (bfs(s, t)) {
        memset(iter, 0, sizeof(iter));
        int f;
        while ((f = dfs(s, t, INF)) > 0) {
            flow += f;
        }
    }
    return flow;
}
```

### 2.3 二分图匹配

**匈牙利算法**：

```cpp
#include <iostream>
#include <cstring>
#include <vector>
using namespace std;

const int MAXN = 505;

vector<int> graph[MAXN];
int match[MAXN];      // match[v] = u 表示 v 匹配了 u
bool visited[MAXN];
int n, m;  // 左右部点数

// 为 u 找匹配
bool dfs(int u) {
    for (int v : graph[u]) {
        if (visited[v]) continue;
        visited[v] = true;
        if (match[v] == -1 || dfs(match[v])) {
            match[v] = u;
            return true;
        }
    }
    return false;
}

// 匈牙利算法
int hungarian() {
    memset(match, -1, sizeof(match));
    int result = 0;
    for (int u = 1; u <= n; u++) {
        memset(visited, false, sizeof(visited));
        if (dfs(u)) result++;
    }
    return result;
}
```

### 3.1 树上问题

**LCA（最近公共祖先）- 倍增法**：

```cpp
const int MAXN = 100005;
const int LOG = 20;

vector<int> tree[MAXN];
int parent[MAXN][LOG];  // parent[i][j] 表示 i 的 2^j 级祖先
int depth[MAXN];

// DFS 预处理
void dfs(int u, int p) {
    parent[u][0] = p;
    depth[u] = depth[p] + 1;
    for (int i = 1; i < LOG; i++) {
        parent[u][i] = parent[parent[u][i-1]][i-1];
    }
    for (int v : tree[u]) {
        if (v != p) dfs(v, u);
    }
}

// 求 LCA
int lca(int u, int v) {
    if (depth[u] < depth[v]) swap(u, v);
    // 将 u 提升到和 v 同一深度
    for (int i = LOG - 1; i >= 0; i--) {
        if (depth[parent[u][i]] >= depth[v]) {
            u = parent[u][i];
        }
    }
    if (u == v) return u;
    // 一起向上跳
    for (int i = LOG - 1; i >= 0; i--) {
        if (parent[u][i] != parent[v][i]) {
            u = parent[u][i];
            v = parent[v][i];
        }
    }
    return parent[u][0];
}
```
