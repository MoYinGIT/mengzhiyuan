## 第10题：综合应用

**程序代码**：
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    srand(time(0));
    
    // 生成10个不重复的随机数（简化算法）
    int nums[10];
    bool used[100] = {false};
    int count = 0;
    
    while (count < 10) {
        int num = rand() % 100;
        if (!used[num]) {
            used[num] = true;
            nums[count] = num;
            count++;
        }
    }
    
    cout << "10个不重复的数: ";
    for (int i = 0; i < 10; i++) {
        cout << nums[i] << " ";
    }
    cout << endl;
    
    // 简单排序（选择排序）
    for (int i = 0; i < 9; i++) {
        for (int j = i + 1; j < 10; j++) {
            if (nums[j] < nums[i]) {
                swap(nums[i], nums[j]);
            }
        }
    }
    
    cout << "排序后: ";
    for (int i = 0; i < 10; i++) {
        cout << nums[i] << " ";
    }
    cout << endl;
    
    return 0;
}
```
