```cpp
    string pwd = "Hello";
    string encrypted = "";
    
    // 加密：每个字母后移3位
    for (int i = 0; i < pwd.length(); i++) {
        char c = pwd[i];
        if (c >= 'a' && c <= 'z') {
            c = c + 3;  // 后移3位
            if (c > 'z') c = c - 26;  // 循环
        } else if (c >= 'A' && c <= 'Z') {
            c = c + 3;
            if (c > 'Z') c = c - 26;
        }
        encrypted += c;
    }
    
    cout << "原文:" << pwd << endl;
    cout << "密文:" << encrypted << endl;
    
    // 解密
    string decrypted = "";
    for (int i = 0; i < encrypted.length(); i++) {
        char c = encrypted[i];
        if (c >= 'a' && c <= 'z') {
            c = c - 3;  // 语句1
            if (c < 'a') c = c + 26;  // 循环处理
        }
        decrypted += c;
    }
    cout << "解密:" << decrypted << endl;
    
    return 0;
}
```

| 整数 → 数字字符 | `ch = n + '0'` | `5 + '0' = '5'` |
| 小写 → 大写 | `upper = lower - 32` | `'a' - 32 = 'A'` |
| 大写 → 小写 | `lower = upper + 32` | `'A' + 32 = 'a'` |

## 核心要点总结

### ASCII编码检查清单

- [ ] 数字字符'0'~'9'的ASCII码是48~57
- [ ] 大写字母'A'~'Z'的ASCII码是65~90
- [ ] 小写字母'a'~'z'的ASCII码是97~122
- [ ] 字符比较就是比较ASCII码
- [ ] 数字字符转整数：`ch - '0'`
- [ ] 整数转数字字符：`n + '0'`
- [ ] 小写转大写：`-32`
- [ ] 大写转小写：`+32`
- [ ] 字符运算本质是ASCII码整数运算
- [ ] 判断字符类型使用ASCII码范围

### 常见错误对比

| 错误写法 | 问题 | 正确写法 |
|:---|:---|:---|
| `int n = '5';` | n=53，不是5 | `int n = '5' - '0';` |
| `char ch = 5;` | ASCII码5，不是'5' | `char ch = '5';` |
| `char c = 5 + '0';` | c='5'，正确但易混淆 | `char c = 5 + '0';` |
| `'a' + 32` | 变成非字母 | `'a' - 32` |
| `'A' - 32` | 变成非字母 | `'A' + 32` |

6. **字符分类**：判断数字、大写、小写
